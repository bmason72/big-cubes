\begin{talltblr}[caption={Improvements provided by the Wideband Sensitivity Upgrade},label=tab:improvements]{
  colspec={ll},
  width=\textwidth,
  hlines,
  vlines,
  row{1} = {bg=headercolor, font=\bfseries}
  }
   Capability &
   Improvement\\
Instantaneous Bandwidth & 
\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Factor of 2 to 4 increase in the {\em available} instantaneous bandwidth (16 to 32\,GHz per polarization) compared to existing receivers.
\end{itemize}
\end{minipage}\\
%
Correlated bandwidth & 
\vspace*{0.1cm}\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Factor of 4 to 68 increase in the {\em correlated} bandwidth at high spectral resolution (0.1--0.2\kms), with larger gains in the lower frequency bands.
\item Observers will no longer need to trade off high spectral resolution for bandwidth.
\end{itemize}
\end{minipage}\\
%
Spectral scan speed   &
\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Increase of at least a factor of 2, and up to a factor of 4 (Band 10) to 64 (Band 1) for a spectral resolution of 0.1--0.2\kms.
\end{itemize}
\end{minipage}\\
%
Spectral line imaging speed         & 
\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Increased spectral line speed from lower receiver noise temperatures ($\sim20$\%; up to $\sim50\%$ at edge of RF band in some receivers), improved digital efficiency ($\sim20$\%), and upgrade to 2SB mixers (Bands 9 and 10 only).
\item Net gain in spectral line imaging speed of $\sim2\text{--}3$.
\end{itemize}
\end{minipage}\\
%
Continuum imaging speed         & 
\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Increase by at least a factor of 3 with 2$\times$ bandwidth increase and at least 6 for 4$\times$ bandwidth, including digital efficiency improvements.
\item Additional gains from improved receiver temperatures.
\end{itemize}
\end{minipage}\\
%
Ultra-high spectral resolution  &
\begin{minipage}{11.2cm}
\begin{itemize}[leftmargin=0.3cm]
\item Provide for the first time unique access to ultra-high spectral resolution observations --- better than 0.01\kms\ at all ALMA frequencies.
\end{itemize}
\end{minipage}
\end{talltblr}
